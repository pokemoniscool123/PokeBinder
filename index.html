  <script>

    const cards = [

      {

        name: "Charizard",

        front: "images/card1-front.jpg",

        back: "images/card1-back.jpg",

        value: "250"

      },

      {

        name: "Pikachu VMAX",

        front: "images/card2-front.jpg",

        back: "images/card2-back.jpg",

        value: "40"

      }

      // Add more cards here later

    ];


    const cardGrid = document.getElementById("cardGrid");

    const popup = document.getElementById("popup");

    const popupName = document.getElementById("popupName");

    const popupFront = document.getElementById("popupFront");

    const popupBack = document.getElementById("popupBack");

    const popupValue = document.getElementById("popupValue");


    cards.forEach(card => {

      const div = document.createElement("div");

      div.className = "card";

      div.innerHTML = `<img src="${card.front}" alt="${card.name}">`;

      div.onclick = () => {

        popupName.textContent = card.name;

        popupFront.src = card.front;

        popupBack.src = card.back;

        popupValue.textContent = card.value;

        popup.style.display = "flex";

      };

      cardGrid.appendChild(div);

    });


    function closePopup() {

      popup.style.display = "none";

    }

  </script>
